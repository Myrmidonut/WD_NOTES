.removeClass();               remove class
.remove();                    remove element

.appendTo();                  move one element to another element
.clone();                     copy an element
.clone().appendTo();          copy an element to another element

.parent();                    access the parent of an element
.children();                  access the children of an element

.css("property", "value");    change css
.prop();                      change html properties
.html();                      change tags and text
.text();                      change only text

-----
SELECTION:
  $("...")            needs css elements, #, ., returns a list, [..., ...]
    div.test          div with class test
    ...:first         first tag of a kind
    ...:last          last tag of a kind
    .parent()         selects the enclosing tag
-----
EXAMPLE:
  $("element").command(function() {...};          selects an element and runs a command on it
-----
EVENTS:
  $("element")
    .click(function() {...})                      runs on mouseclick
    .keypress(function() {...})                   runs on keypress
    .on("...", function() {...})                  works with listeners on future elements
    $("ul").on("click", "li", function(){...})    example, on has to run on ul for future lis
      "click"
      "keypress"
      "mouseenter"
      "mouseleave"
    .hover(function() {...})                      mouseover

  $(this)                                         refers to the one element that is accessed
-----
MANIPULATION:
  $("element")
    .html()                 gets the whole HTML, see innerHTML
    .html("...")            changes the whole HTML
    .attr("...")            get the value of an attribute
    .attr("...", "...")     changes an attribute
    .css("...")             gets the value of the styling
    .css("...", "...")      changes styling
    .text()                 gets content of a tag, see text-content
    .text("...")            changes content of a tag
    .val()                  gets value of an element, see value
    .val("...")             changes value of an element

    .addClass("...")        adds a class
    .removeClass("...")     removes a class
    .toggleClass("...")     toggles a class

    .remove()               removes an element from the page
    .append("...")          adds text to the end of an element
-----
EFFECTS:
  $("element")
    .fadeOut(duration, function(){})        fade out effect, optional duration in ms, optional function after fadeout
      "slow", a number, etc.                values for duration
                                            elements stay as hidden on the page
    .fadeIn(duration, function(){})         fade in effect
    .fadeToggle(duration, function(){})     toggled fade effect

    .hide()                                 removes the element from the flow

    .slideDown()                            element gets squished in height
    .slideUp()                              element gets increased in height
    .slideToggle()                          toggles slide

    .animate({                              animates an element, duration in ms, optional function
      property1: "value1",
      property2: "value2"
    }, duration, function(){})
