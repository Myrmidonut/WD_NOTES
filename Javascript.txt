ARRAYS:                                 ordered list of data
  var myArray = [];                     creates an empty array
  var myArray = new Array();            creates an empty array
  var myArray = [1, 2, 3];              creates an array
    myArray[0] = ...;
    myArray[3][4]                       nested array
    myArray.length;                     number of entries
    myArray.push(...);                  adds an entry to the end, returns the new length
    myArray.pop();                      removes the last entry, returns the removed entry
    myArray.unshift(...);               adds an entry to the beginning, returns the new length
    myArray.shift();                    removes the first entry
    myArray.splice(a, b);               removes an entry, starts at a and deletes b entries
    myArray.splice(a, 0, b);            add an entry at position a with the content b
    myArray.splice(a, 1, b);            replace an entry at position a with content of b
    myArray.indexOf(...);               returns the first index of that entry, -1 if not exists
  var myArray2 = myArray.slice(1, 3);   copies parts of an array into a new one, 
                                        starts at 1 and ends one before 3
  var myArray2 = myArray.slice();       copies the whole array
    myArray.forEach(function(name) {    looping over the entries in an array
  ...                                   and runs the function for the entries
    } );                                passed in via the variable "name"

USEFUL FUNCTIONS:
  Math.floor(...)                     removes the decimals
  Math.random()                       random number between 0 and 1, e.g. 0.123, but not 1
    Math.random() * 5                 random number between 0 and 5, e.g. 2.123, but not 5!
    Math.random() * 5 + 1             random number between 1 and 5, including 1 and 5
  Math.floor(Math.random() * 6)       random number out of 1, 2, ..., 5

  setTimeout(function, ms)            execution after time in ms
