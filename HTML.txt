    <article>...</article>

ENTITIES:			display symbols, etc.
  &lt;				< 
  &gt;				>
  &amp;				&
  &quot;			"
  &apos;			'
  &euro;			€
  &nbsp;			blank space, non-breaking space
-----
FORMATING:
  <h1>...</h1>			Header 1-6
  <br>					linebreak
  <p>...</p>			paragraph, linebreak
  <strong>...</strong>	bold, preferable to
    <b>...</b>
  <em>...</em>			italic, preferable to
    <i>...</i>
  <ins>...</ins>		underlined, preferable to
    <u>...</u>
  <sup>...</sup>		supscript (hochgestellt)
  <sub>...</sub>		subscript
  <del>...</del>		crossed out
  <hr>					line
-----
LISTS:
  <ul>...</ul>			unordered list, points
    <li>...</li>		entries
  <ol>...</ol>			ordered list, numbers
    <li>...</li>		entries
  <ol start="10">		starts at 10
  <ol reversed>			reversed list
  <ol type="1">			symbols: 1, A, I, a, i
  						lists can be inside another list
-----
IMAGES:
  <img src="...">		image, local file or link
    width="100"			scales width to 100 pixels
    height="100"		scales height to 100 pixels
    align="left"		left of website
    align="right"		right of website
    <p></p>				text under image
    alt="..."			adds a description to the image for search engines, visually impared users, etc.
-----
VIDEOS:
  <video src="...">...</video>	link to video file, ... only shows when the video can not be loaded
    width="..."					width
    height="..."				height
    controls					show pause, play and skip
-----
FORMS:
  <form>...</form>		form, linebreak			
    action="..."		URL to send data to			
    method="post"		type of HTTP request
    method="GET"

    <input>						input box, default is text
      type="text"				text
      type="number"				number
      type="password"			password, hidden
      type="email"				email
      type="date"				date
      type="color"				color
      type="file"				file
      type="checkbox"			checkbox
        checked					default checked for cbox
      type="radio" name="..."	radio button, linked with "...", only one available
      type="submit"				submit button	
   
      placeholder="..."			background writing in text, number, etc
      pattern=".{5,10}"			min and max length of input
      title="5 to 10"			error code for pattern
      required					checks for the correct input

    <textarea rows="10" cold="10">...</textarea>	more boxes than "text"

    <select>...</select>			dropdown
      <option>...</option>			entries
        selected					default selected

      name="..."					name for the data that will be send, shows up in URL as name=value
      value="..."					data that will be send

    <label for="...">...</label>	label for a form, linked with for="..."
    <input id="...">				for accessibilty, for= and id= has to be equal

  EXAMPLE:
    <div>
      <label for="first_name">First Name</label>
      <input name="first_name" type="text">
    </div>

LINKS:
  <a href="...">...</a>		links, Text is clickable, local file or link
    <p id="...">
    href="./file.html"		./ represents the current folder
    href="#..."				leads to an id="..." on the page
    target="_blank"			opens a new tab
