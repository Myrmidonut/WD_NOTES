ODM:
  Object Data Mapper
---------------------------------------------------------------------------------------------------
INSTALL:
  npm install mongoose --save
---------------------------------------------------------------------------------------------------
USE:
  var mongoose = require("mongoose");
---------------------------------------------------------------------------------------------------
CONNECT:
  mongoose.connect("mongodb://localhost/...");      connects to database or creates it first
---------------------------------------------------------------------------------------------------
SCHEMA:
  var catSchema = new mongoose.Schema({
    name: String,
    age: Number,
    temperament: String
  });
---------------------------------------------------------------------------------------------------
MODEL:
  var Cat = mongoose.model("Cat", catSchema);
---------------------------------------------------------------------------------------------------
CREATE:
  Cat.create({
    name: "Snow White",
    age: 15,
    temperament: "Bland"
  }, function(err, cat) {
    if (err) {
      console.log(err);
    } else {
      console.log(cat);
    }
  });
---------------------------------------------------------------------------------------------------
FIND:
  Cat.find({}, function(err, cats) {
    if (err) {
      console.log(err);
    } else {
      console.log(cats);
    }
  });
